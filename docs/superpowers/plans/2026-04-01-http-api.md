> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add an optional HTTP API server to Jackdaw for remote session monitoring, with REST endpoints, SSE streaming, and bearer token auth.

**Architecture:** A new `http.rs` module spawns an axum server alongside the existing IPC socket server. Both share `Arc<AppState>`. REST handlers call the existing `api::handle_query`/`api::handle_action` functions. SSE subscribes to the existing `state.subscriber_tx` broadcast channel. Auth via auto-generated bearer token stored at `~/.jackdaw/api-token`.

**Tech Stack:** Rust, axum, axum (sse feature), tower-http (cors), tokio, rand

```rust
use std::path::Path;

/// Load token from file, or generate a new one if it doesn't exist.
/// Returns the hex-encoded token string.
pub fn load_or_generate_token(token_path: &Path) -> std::io::Result<String> {
    if token_path.exists() {
        let token = std::fs::read_to_string(token_path)?.trim().to_string();
        return Ok(token);
    }

    use rand::Rng;
    let mut rng = rand::thread_rng();
    let bytes: [u8; 32] = rng.gen();
    let token: String = bytes.iter().map(|b| format!("{:02x}", b)).collect();

    if let Some(parent) = token_path.parent() {
        std::fs::create_dir_all(parent)?;
    }
    std::fs::write(token_path, &token)?;

    #[cfg(unix)]
    {
        use std::os::unix::fs::PermissionsExt;
        std::fs::set_permissions(token_path, std::fs::Permissions::from_mode(0o600))?;
    }

    Ok(token)
}
```

```rust
use crate::state::AppState;
use std::sync::Arc;

/// Start the HTTP API server if enabled in config.
/// This function reads config from the Tauri store and returns immediately
/// if the HTTP API is disabled.
pub async fn start_http_server(state: Arc<AppState>, app_handle: tauri::AppHandle) {
    use tauri_plugin_store::StoreExt;

    let config = app_handle
        .store("settings.json")
        .ok()
        .and_then(|store| store.get("http_api"))
        .and_then(|v| serde_json::from_value::<HttpConfig>(v).ok())
        .unwrap_or_default();

    if !config.enabled {
        eprintln!("Jackdaw: HTTP API disabled");
        return;
    }

    let token_path = dirs::home_dir()
        .expect("could not determine home directory")
        .join(".jackdaw")
        .join("api-token");

    let token = match load_or_generate_token(&token_path) {
        Ok(t) => Arc::new(t),
        Err(e) => {
            eprintln!("Jackdaw: failed to load/generate API token: {}", e);
            return;
        }
    };

    let addr: std::net::SocketAddr = format!("{}:{}", config.bind_address, config.port)
        .parse()
        .unwrap_or_else(|_| {
            eprintln!("Jackdaw: invalid bind address, falling back to 127.0.0.1:{}", config.port);
            std::net::SocketAddr::from(([127, 0, 0, 1], config.port))
        });

    eprintln!("Jackdaw: HTTP API listening on {}", addr);

    let app = build_router(state, token);

    let listener = match tokio::net::TcpListener::bind(addr).await {
        Ok(l) => l,
        Err(e) => {
            eprintln!("Jackdaw: failed to bind HTTP server: {}", e);
            return;
        }
    };

    if let Err(e) = axum::serve(listener, app).await {
        eprintln!("Jackdaw: HTTP server error: {}", e);
    }
}
```

```rust
use axum::{
    Router,
    extract::State as AxumState,
    http::{Request as AxumRequest, StatusCode},
    middleware::{self, Next},
    response::{IntoResponse, Json},
};

#[derive(Clone)]
struct HttpState {
    app_state: Arc<AppState>,
    token: Arc<String>,
}

fn build_router(state: Arc<AppState>, token: Arc<String>) -> Router {
    let http_state = HttpState {
        app_state: state,
        token,
    };

    let authed_routes = Router::new()
        .fallback(fallback_handler)
        .route_layer(middleware::from_fn_with_state(
            http_state.clone(),
            auth_middleware,
        ));

    let public_routes = Router::new()
        .route("/api/health", axum::routing::get(health_handler));

    public_routes
        .merge(authed_routes)
        .with_state(http_state)
}

async fn auth_middleware(
    AxumState(state): AxumState<HttpState>,
    req: AxumRequest<axum::body::Body>,
    next: Next,
) -> impl IntoResponse {
    let auth_header = req
        .headers()
        .get("authorization")
        .and_then(|v| v.to_str().ok());

    match auth_header {
        Some(header) if header.starts_with("Bearer ") => {
            let provided = &header[7..];
            if provided == state.token.as_str() {
                next.run(req).await.into_response()
            } else {
                (StatusCode::UNAUTHORIZED, Json(serde_json::json!({"error": "invalid or missing token"}))).into_response()
            }
        }
        _ => {
            (StatusCode::UNAUTHORIZED, Json(serde_json::json!({"error": "invalid or missing token"}))).into_response()
        }
    }
}

async fn health_handler() -> impl IntoResponse {
    Json(serde_json::json!({"ok": true}))
}

async fn fallback_handler() -> impl IntoResponse {
    (StatusCode::NOT_FOUND, Json(serde_json::json!({"error": "not found"})))
}
```

```rust
async fn subscribe_handler(

) -> Sse<impl tokio_stream::Stream<Item = Result<Event, std::convert::Infallible>>> {
```
